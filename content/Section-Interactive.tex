library(rgl)
data(quakes)
plot3d(
	x=quakes$long, 
	y=quakes$lat, 
	z=quakes$depth, 
	xlab="Longitude", 
	ylab="Latitude", 
	zlab="Depth"
	)
\end{lstlisting}
