\subsection{Adding Hyperlinks and Buttons}
% ------------------------------------------------------------

  \begin{itemize}
  \item<1-> First item.
  \item<2-> Second item.
  \item<3-> Third item.
  \end{itemize}

  \hypertarget<2>{jumptosecond}{}
\end{frame}

\frame{
  \begin{theorem}
    ...
  \end{theorem}
  \begin{overprint}
  \onslide<1>
    \hfill\hyperlinkframestartnext{\beamerskipbutton{Skip proof}}
  \onslide<2>
    \begin{proof}
      ...
    \end{proof}
  \end{overprint}
}

 {
   \begin{theorem}

   \end{theorem}

   \onslide<2>
     \begin{proof}
       ...
     \end{proof}
     \hfill\hyperlink{mytheorem<1>}{\beamerreturnbutton{Return}}
   \end{overprint}
 }
